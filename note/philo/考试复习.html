def generate_html_from_text(text):
    # 将文本按行分割
    lines = text.split('\n')

    # 用于存储生成的HTML代码
    html = []

    # 添加HTML头部
    html.append('<!DOCTYPE html>')
    html.append('<html>')
    html.append('<head>')
    html.append('<title>答题界面</title>')
    html.append('<style>')
    html.append('body {font-family: Arial, sans-serif;}')
    html.append('.question {margin-bottom: 10px;}')
    html.append('.result {font-weight: bold; margin-top: 5px;}')
    html.append('.correct {color: green;}')
    html.append('.incorrect {color: red;}')
    html.append('</style>')
    html.append('</head>')
    html.append('<body>')
    html.append('<h1>答题界面</h1>')

    question_count = 0
    correct_count = 0
    for line in lines:
        line = line.strip()
        if line:
            if line.startswith(str(question_count + 1) + '.'):
                # 处理问题行
                question = line.split(str(question_count + 1) + '.')[1].strip()
                html.append('<div class="question">')
                html.append('<p>{}</p>'.format(question))
            elif line.startswith('My Answer:'):
                # 处理用户答案行
                user_answer = line.split(':')[1].strip()
                html.append('<p class="result">你的答案: <span class="{}">{}</span></p>'.format('correct' if user_answer == 'Correct Answer' else 'incorrect', user_answer))
                html.append('</div>')
                question_count += 1
                if user_answer == 'Correct Answer':
                    correct_count += 1
    

    html.append('</body>')
    html.append('</html>')

    return '\n'.join(html)
