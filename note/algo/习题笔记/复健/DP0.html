int val[105][105], dp[105][105], fa[105][105];

void print(int x, int y){
    if(x==0) return; 
    if(fa[x][y] == y){
        printf("%d ", y);
        return;
    }
    print(x-1, fa[x][y]);
    printf("%d ", y);

}

int main(){
    F(i, 1, 104) F(j, 1, 104) dp[i][j]=-1*0x3f3f3f3f;
    int f, v; cin>>f>>v;
    F(i, 1, f) F(j, 1, v) cin>>val[i][j]; 
    F(i, 1, v-f+1){
        dp[1][i] = val[1][i];
    }

    F(i, 1, f-1) F(j, i+1, v-f+i+1) F(k, i, j-1){
        // dp[i+1][j] =  max(dp[i][k]+val[i+1][j], dp[i+1][j]);
        if(dp[i][k]+val[i+1][j] > dp[i+1][j]){
            dp[i+1][j] =  dp[i][k]+val[i+1][j];
            fa[i+1][j] = k;
        }

    }
    int ans = -1;
    int start;
    F(i, 1, v){
        if(ans<dp[f][i]){
            ans = dp[f][i];
            start = i;
        }
    }
    cout<<ans<<endl;
    print(f, start);    
}

int dfs(int x, int t){
    debugf("Now %d %d\n", x,t );
    if(x > w || x<=0) return 0;
    if(t < 0) return 0;
    int ans = max({
        dfs(x, t-1),
        dfs(x-1, t-1),
        dfs(x-2, t-1),
        dfs(x+1, t-1),
        dfs(x+2, t-1),  
    })+val[x][t];
    debug(ans);
    return ans;
}

int a, b, c, d;

    cin>>h>>w;
    h--;
    while(cin>>a>>b>>c>>d){

}
