A. 排序

我们可以把最大的放在一个连续区间里面, 因此就枚举每个区间, 看最大的取和就行了.

int a[MAXN];
void solve(){
    int n, k;
    cin>>n>>k;
    int ans = 0;
    F(i, 1, n) cin>>a[i];
    for(int i=1; i<=k; i++){
        int cur = i; 
        int mx = 0;
        while(cur<=n){
            // printf("a[%d] = %d\n", cur,a[cur]);

            mx = max(mx, a[cur]);
            cur += k;
        }
        // printf("Ans += %d\n", mx);
        ans += mx;
    }
    cout<<ans<<endl;
}

B. 构造

每次会出来一个胜者, 一个输家. 并且总有一个输家的赢得个数是0. 因此就可以着手构造0 m这种情况就行了.

void solve(){
    int n, a, b; cin>>n>>a>>b;
    int mx = max(a, b);
    if(mx == 0) {
        printf("-1\n");
        return;
    }
    if(min(a, b) != 0 || (n-1)% mx ){
        printf("-1\n");
        return;
    }
    for(int i=2; i<=n; i+=max(a,b)){
        F(j, 1, max(a, b)){
            printf("%d ", i);
        } 
    }
    printf("\n");
}

C. 构造

最多 \(n\) 次, 肯定用满就可以构造了. 因为 \(l\) 和 \(r\) 是可以交换的, 让他们都相等也不是难事.

int a[MAXN];
void solve(){
    int n; cin>>n;
    F(i, 1, n) cin>>a[i];
    
    printf("%d\n", n-1);
    if(n>1) printf("1 %d\n", n);

    int par = (a[1]+a[n])%2 ? a[1]:a[n];
    F(i, 2, n-1){
        if((par+a[i])%2) printf("1 %d\n", i);
        else printf("%d %d\n", i,n);
    }


}

D1. 构造

就是看到有多少个不一样的数字, 然后如果是奇数个就无解, 偶数个的话:

  • 如果是2个就直接 \(\min(2x, y)\)
  • 多于两个就可以尽量用 \(y\) , 然后直接加就行了.
vector<pair<int, int>> p;
void solve(){
    p.clear();
    int n, x, y; cin>>n>>x>>y;
    string s, t;
    cin>>s; cin>>t;
    s = "#"+s; t="#"+t;
    int diff = 0;
    F(i, 1, n){
        if(s[i] != t[i]){
            diff++;
            p.push_back(make_pair(i, s[i]-'0'));
        }
    }
    int ans = 0;
    int l = 0, r = p.size()-1;
    if(diff==0){printf("0\n"); return;}
    if(diff==2){printf("%d\n", (p[l].fi+1 == p[r].fi)?min(2*y,x):y); return;}
    if(diff % 2){printf("-1\n"); return;}
    while(r-l+1>2){
        ans += y;
        r--, l++;
    }
    ans += y;
    cout<<ans<<endl;

}
