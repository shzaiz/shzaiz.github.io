A. 数组越界

注意循环变量的范围.

int a[MAXN], b[MAXN];
vector<int > fi, fo; 
void solve(){
    fi.clear();
    fo.clear();
    int n; cin>>n;
    F(i, 1, n) cin>>a[i];
    F(i, 1, n) cin>>b[i];
    F(i, 1, n){
        if(a[i] == 1){
            fi.push_back(b[i]);
        }else{
            fo.push_back(b[i]);
        }
    }
    
    sort(fi.begin(), fi.end());
    sort(fo.begin(), fo.end());
    reverse(ALL(fi));
    reverse(ALL(fo));
    // for(auto i: fi) cout<<i<<" ";
    // printf("\n");
    // for(auto i: fo) cout<<i<<" ";
    // printf("\n====\n");
    int al=fi.size(), bl=fo.size();
    int minn = min(al, bl);
    int ans = 0;
    if(al == 0){
        F(i, 0, bl-1){
            ans += fo[i];
        }
        cout<<ans<<endl;
    }else if(bl==0){
        F(i, 0, al-1){
            ans += fi[i];
        }
        cout<<ans<<endl;
    }else if(al==bl){
        F(i, 0, minn-1){
            if(i==minn-1){
                ans += max(fi[i]*2+fo[i], fi[i]+fo[i]*2);
                break;
            }
            ans += (fi[i]+fo[i])*2;
        }
        cout<<ans<<endl;
    }else{
        // printf("Invoke");
        F(i, 0, minn-1){
            ans += (fi[i]+fo[i])*2;
            
        }
        F(i, minn, (al>bl?al:bl)-1){
            ans += (al>bl?fi[i]:fo[i]);
        }
        cout<<ans<<endl;
    }
    // printf("---===---\n");
}

B. 分类讨论

注意分大多数类讨论, 以及想清楚什么时候应当NO. 如果大小写不计较的话可以用YES, YeS, yES来区别自己的分类讨论结果.

int a[MAXN];
void solve(){
    int n, k; cin>>n>>k;
    F(i, 1, k) cin>>a[i];
    bool flag = 1;
    int currd = LONG_LONG_MAX;
    Fd(i, k, 2){
        int ai = a[i]-a[i-1];
        if(ai <= currd){
            currd = ai;
        }else{
            flag = 0;
            break;
        }
    }
    if(!flag) {
        printf("No\n");
        return;
    }else{
        if(k==1){
            printf("YeS\n");
            return;
        }else if(n-k==0){
            if(a[1]>currd){
                printf("No\n"); return;
            }else{
                printf("Yes\n"); return;
            }
        }else {
            int len = (n-k+1);
            
            if(len*currd>=a[1]) {
                printf("YES\n");
                return;
            }else{
                printf("NO\n");
                return;

            }
        }
        printf("Yes\n");
    }

}
