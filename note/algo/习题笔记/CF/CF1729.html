这个题还是挺需要一点分类讨论的.

signed main(){
    #ifdef XBZAKIOI

void solve(){
    string s;cin>>s; 
    int n = s.size();
    map<char, vector<int>> l;
    F(i, 0, n-1) l[s[i]].push_back(i);
    int d = (s[0]<s[n-1] ? 1 : -1);
    vector<int> ans;
    for(char i= s[0]; i != s[n-1]+d; i+=d){
        for(auto now:l[i]){
            ans.push_back(now);
        }
    }
    int co = 0;
    F(i, 1, ans.size()-1) 
        co += abs(s[ans[i]]-s[ans[i-1]]);

    cout<< co<<" "<<ans.size()<<endl;

    for (auto i : ans){
        cout<<i+1<<" ";
    }
    cout<<endl;
}

D. 构造贪心(c)

让最大的和最小的配对就可以了. 因为最多2个人, 这样做是最优的. 可以考虑用一个双指针来完成.

其实有时候也不用想太多, 毕竟数组已经有序了, 因此就是要用好这个性质.

int d[MAXN];
bool vis[MAXN];
int a[MAXN], b[MAXN];
void solve(){
    int n;
    cin>>n;
    F(i, 1, n) cin>>a[i];
    F(i, 1, n) cin>>b[i];

    F(i, 1, n){
        d[i] = b[i]-a[i];
    }

    sort(d+1, d+1+n);

    // F(i, 1, n) printf("%d ", d[i]);

    int i = 1, j = n;
    int cnt = 0;
    while(i<j){
        while(d[i]+d[j]<0 && i<j) i++;
        if(i<j)cnt++;
        i++, j--;
        
    }
    cout<<cnt<<endl;
}
