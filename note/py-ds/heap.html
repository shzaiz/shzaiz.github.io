Priority Queue using heap

Complete binary tree: Full size, left to right like.

class Heap:
    def __init__(self, key=lambda x:x):
        self.data = []
        self.key  = key

    @staticmethod
    def _parent(idx):
        return (idx-1)//2
        
    @staticmethod
    def _left(idx):
        return idx*2+1

    @staticmethod
    def _right(idx):
        return idx*2+2
    
    def heapify(self, idx=0):
        while True:
            l = Heap._left(idx)
            r = Heap._right(idx)
            
            pk = idx
            le = len(self.data)
            if( l<le and
                self.key(self.data[l]) > self.key(self.data[pk])):
                pk = l 
            if( r<le and
                self.key(self.data[r]) > self.key(self.data[pk])):
                pk = r
            
            if(pk == idx):
                break
            else:
                self.data[idx],self.data[pk]=self.data[pk],self.data[idx]
                idx=pk

            
    def add(self, x):
        self.data.append(x)
        cidx = len(self.data)-1
        cprt = Heap._parent(cidx)
        while(cidx>0 and self.key(self.data[cidx])>self.key(self.data[cprt])):
            self.data[cidx], self.data[cprt] = self.data[cprt], self.data[cidx]
            cidx = cprt
            cprt = Heap._parent(cidx) 
        
    def peek(self):
        return self.data[0]

    def pop(self):
        ret = self.data[0]
        self.data[0] = self.data[len(self.data)-1]
        del self.data[len(self.data)-1]
        self.heapify()
        return ret
    
    def __bool__(self):
        return len(self.data) > 0

    def __len__(self):
        return len(self.data)

    def __repr__(self):
        return repr(self.data)

Application: Running Median

We shall use this kind of idea, to generate a max and a min heap, then control their bypasses. Making the delta less than 1.

def running_medians(it):
    med = None
    mxh = Heap(lambda x: x)
    mih = Heap(lambda x: -x)
    ans = []
    for i in it:
        # print(i)
        if(med == None):
            mxh.add(i)
            med = i
            ans.append(med)
            continue
        
        if(i<med):
            mxh.add(i)
        elif(i>=med):
            mih.add(i)
        
        if(len(mxh)-len(mih)>1):
            mih.add(mxh.pop())  
        if(len(mih)-len(mxh)>1):
            mxh.add(mih.pop())

        if(len(mxh) == len(mih)):
            med = (mxh.peek()+mih.peek())/2
        elif(len(mxh)>len(mih)):
            med = mxh.peek()
        else:
            med = mih.peek()
        
        ans.append(med)
    # print(ans)
    return ans
