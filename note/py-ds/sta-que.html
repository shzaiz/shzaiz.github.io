prec = {'*': 2, '/': 2,
        '+': 1, '-': 1}

def infix_to_postfix(expr):
    """Returns the postfix form of the infix expression found in `expr`"""
    ops = Stack()
    postfix = []
    toks = expr.split()
    for tok in toks:
        if tok.isdigit():
            postfix.append(tok)
        elif tok in delim_openers:
            ops.push(tok)
        elif tok in delim_closers:
            while ops and ops.peek() != delim_openers[delim_closers.index(tok)]:
                postfix.append(ops.pop())
            ops.pop()
        elif tok in prec:
            while ops and ops.peek() != '(' and prec[ops.peek()] >= prec[tok]:
                postfix.append(ops.pop())
            ops.push(tok)
    while ops:
        postfix.append(ops.pop())

            
    return ' '.join(postfix)

Circular queue

class Queue:
    def __init__(self, limit=10):
        self.data = [None] * limit
        self.head = -1
        self.tail = -1

    def nxt(self, id, len):
        return (id+1)%len

    def enqueue(self, val):
        # We make sure that after it is init, no -1 are allowed
        if(self.tail == -1 or self.head == -1):
            self.tail = 0
            self.head = 0
            self.len = 1
            self.data[0] = val
            return 
        l = len(self.data)
        nhd = self.nxt(self.head, l)
        if(nhd == self.tail):
            raise RuntimeError("The Q is full!")
        self.head = nhd 
        self.data[nhd] = val
        self.len += 1
        
        
    def dequeue(self):
        if(self.tail == -1 or self.head == -1):
            raise RuntimeError("The Q is empty!")
        l = len(self.data)
        ntl = self.nxt(self.tail, l)
        if(ntl == self.tail):
            raise RuntimeError("The Q is empty!")
        ret = self.data[self.tail]
        self.data[self.tail] = None
        self.tail = ntl 
        self.len -=1
        if(self.len == 0):
            self.head = -1
            self.tail = -1
        return ret
        

    
    def resize(self, newsize):
        assert(len(self.data) < newsize)
        l = self.len
        nhd = self.len-1
        ntl = 0
        nlst = [None]*newsize

        for i in range(l):
            nlst[i] = self.data[(self.tail+i)%self.len]
        self.data = nlst
        self.head = nhd
        self.tail = ntl
		
    
    def empty(self):
        return (self.head == -1 or self.len == 0)
    
    def __bool__(self):
        return not self.empty()
    
    def __str__(self):
        if not(self):
            return ''
        return ', '.join(str(x) for x in self)
    
    def __repr__(self):
        return str(self)
    
    def __iter__(self):
        if(self.head == -1 or self.tail == -1):
            raise StopIteration
        for i in range(self.len):
            yield self.data[(self.tail+i)%self.len]
