定义e为某个点的值, ne是某个点下一个值. 之间是用下标关联出来的.

#include <bits/stdc++.h>
using namespace std;

#define MAXN 100010
int head, idx, e[MAXN], ne[MAXN];

void init(){
  head = -1; idx = 0;
}
void prep(int x){
  e[idx] = x, ne[idx] = head, head = idx++;
}
void add(int k, int x){
  e[idx] = x;
  ne[idx] = ne[k];
  ne[k] = idx++;
}
void rem(int k){
  if(k==-1) {
    head = ne[head];
    return;
  }
  ne[k] = ne[ne[k]];
}
int main(){
  int m; cin>>m; 
  init(); 
  for(int i=1; i<=MAXN-1; i++) ne[i]=-1;
  while(m--){
    char op; cin>>op;
    if(op == 'H'){
      int x; cin>>x;
      prep(x);
    }else if(op == 'D'){
      int x; cin>>x;
      rem(x-1);
    }else{
      int k, x; cin>>k>>x;
      add(k-1,x);
    }
    
  }
  for(int i=head; i!=-1;i=ne[i])cout<<e[i]<<" ";
  return 0;
  
}

单调栈

  1. 解决的问题: 给定一个长度为N的整数数列，输出每个数左边第一个比它小的数，如果不存在则输出-1.

想法: 维护一个储存最小值的栈, 使得其栈顶永远是和前面元素的最小值.

Code.

#include<bits/stdc++.h>
using namespace std;

int n, stk[100010], tt=0;
int main(){
  cin>>n; 
  for(int i=0; i<n; i++){
    int x; cin>>x;
    while(tt && stk[tt]>=x) tt--;
    if(tt) cout<<stk[tt]<<" ";
    else cout<<-1<<" ";
    stk[++tt]=x;
  }
}
