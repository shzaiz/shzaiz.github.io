定义e为某个点的值, ne是某个点下一个值. 之间是用下标关联出来的.

#include <bits/stdc++.h>
using namespace std;

#define MAXN 100010
int head, idx, e[MAXN], ne[MAXN];

void init(){
  head = -1; idx = 0;
}
void prep(int x){
  e[idx] = x, ne[idx] = head, head = idx++;
}
void add(int k, int x){
  e[idx] = x;
  ne[idx] = ne[k];
  ne[k] = idx++;
}
void rem(int k){
  if(k==-1) {
    head = ne[head];
    return;
  }
  ne[k] = ne[ne[k]];
}
int main(){
  int m; cin>>m; 
  init(); 
  for(int i=1; i<=MAXN-1; i++) ne[i]=-1;
  while(m--){
    char op; cin>>op;
    if(op == 'H'){
      int x; cin>>x;
      prep(x);
    }else if(op == 'D'){
      int x; cin>>x;
      rem(x-1);
    }else{
      int k, x; cin>>k>>x;
      add(k-1,x);
    }
    
  }
  for(int i=head; i!=-1;i=ne[i])cout<<e[i]<<" ";
  return 0;
  
}
