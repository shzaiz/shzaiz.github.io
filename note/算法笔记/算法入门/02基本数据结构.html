+---+---+

定义e为某个点的值, ne是某个点下一个值. 之间是用下标关联出来的.

#include <bits/stdc++.h>
using namespace std;

#define MAXN 100010
int head, idx, e[MAXN], ne[MAXN];

void init(){
  head = -1; idx = 0;
}
void prep(int x){
  e[idx] = x, ne[idx] = head, head = idx++;
}
void add(int k, int x){
  e[idx] = x;
  ne[idx] = ne[k];
  ne[k] = idx++;
}
void rem(int k){
  if(k==-1) {
    head = ne[head];
    return;
  }
  ne[k] = ne[ne[k]];
}
int main(){
  int m; cin>>m; 
  init(); 
  for(int i=1; i<=MAXN-1; i++) ne[i]=-1;
  while(m--){
    char op; cin>>op;
    if(op == 'H'){
      int x; cin>>x;
      prep(x);
    }else if(op == 'D'){
      int x; cin>>x;
      rem(x-1);
    }else{
      int k, x; cin>>k>>x;
      add(k-1,x);
    }
    
  }
  for(int i=head; i!=-1;i=ne[i])cout<<e[i]<<" ";
  return 0;
  
}

单调栈

  1. 解决的问题: 给定一个长度为N的整数数列，输出每个数左边第一个比它小的数，如果不存在则输出-1.

想法: 维护一个储存最小值的栈, 使得其栈顶永远是和前面元素的最小值.

Code.

#include<bits/stdc++.h>
using namespace std;

int n, stk[100010], tt=0;
int main(){
  cin>>n; 
  for(int i=0; i<n; i++){
    int x; cin>>x;
    while(tt && stk[tt]>=x) tt--;
    if(tt) cout<<stk[tt]<<" ";
    else cout<<-1<<" ";
    stk[++tt]=x;
  }
}

单调队列(滑动窗口)

  1. 解决的问题: 一个滑动的区间内部的最大值和主次熬制都输出出来.

  2. 暴力: 可以先用一个队列来维护窗口, 保证每次这个窗口里面存的是当前的所有元素. 遍历所有元素, 得到时间的复杂度是 \(\mathcal O(nk)\) .

  3. 考虑优化: 有些元素是没有用的.

Code.

#include <bits/stdc++.h>
using namespace std;

#define MAXN 1000100
// 队列里面存放的是下标
int n,k, a[MAXN], q[MAXN];

int main(){
  scanf("%d%d", &n, &k);
  int hh = 0, tt = -1;
  for(int i=0; i<n;i++){
    scanf("%d", &a[i]);
  }
  for(int i=0; i<n; i++){
    // 判断队头是不是划出了窗口
    if(hh <= tt && i-k+1>q[hh]) hh++;
    while(hh<=tt && a[q[tt]]>=a[i]) tt--;
    q[++tt] = i;
    if(i>=k-1) printf("%d ", a[q[hh]]);
  }
  puts("");
  hh=0, tt=1;
  for(int i=0; i<n; i++){
    // 判断队头是不是划出了窗口
    if(hh <= tt && i-k+1>q[hh]) hh++;
    while(hh<=tt && a[q[tt]]<=a[i]) tt--;
    q[++tt] = i;
    if(i>=k-1) printf("%d ", a[q[hh]]);
  }
  
}

int main(){
  cin>>n>>p+1>>m>>s+1;

  }

Trie树

要求: 高效的存储字符串集合的数据结构

  • 在每个字符的结尾打一个标记(acceptable state)

存储: 从根节点开始, 每次不存在就增加一个, 在最后的内容加一个标记.

查找: 沿着已经有的路径走. 如果不存在/走到最后没有标记, 那么就不存在, 反之存在.

Code.

#include <bits/stdc++.h>
using namespace std;

int son[100010][26], cnt[100010], idx;

void insert(char str[]){
  int p = 0; 
  for(int i=0; str[i]; i++){
    int u = str[i] - 'a';
    if(!son[p][u]) son[p][u] = ++idx;
    p = son[p][u];
  }
  cnt[p]++;
}

int query(char str[]){
  int p=0;
  for(int i=0; str[i]; i++){
    int u = str[i] - 'a';
    if(!son[p][u]) return 0;
    p = son[p][u];
  }
  return cnt[p];
}

int n;
char str[1000010];
int main(){
  scanf("%d", &n);
  while(n--){
    char op[2];
    scanf("%s%s", op, str);
    if(op[0] == 'I') insert(str);
    else printf("%d\n", query(str));
  }
  return 0;
}

并查集

要求:

  • 将两个集合合并
  • 判定两个元素是不是在一个集合中

原理: 每个集合用一棵树来表示, 树根的编号就是整个集合的编号, 每个节点储存着他的父节点的编号, p[x]表示x的父节点.

操作:

  • 判断树根: p[x]==x
  • 求x的集合编号: while(p[x]!=x) x = p[x]
  • 如何合并两个集合px,py, \(x\neq y\) . p[x]=y.

Code.

变形例1(Acw837). 维护每个集合里面的元素个数

解答

首先套用上一题的内容.

int n, m; 
int p[MAXN], size[MAXN];
int find(int x){
    if(p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int main(){
    scanf("%s%d%d", op, &n, &m);
    for(int i=1; i<=n; i++) {
        p[i] = i; size[i] = 1;
    }
    while(m--){
        char op[2]; 
        int a, b;
        scanf("%s", op);
        if(op[0] == 'C'){
            scanf("%d%d", &a, &b);
            if(find(a) == find(b)) continue;
            size[find(b)] += size[find(a)];
            p[find(a)] = find(b);
        }else if(op[1]=='1'){
            scanf("%d%d", &a,&b);
            if(find(a) == find(b)) cout<<"Yes\n"; else cout<<"No\n";
        }else{
            scanf("%d", &a);
            // 保证根节点的size是有意义的, 合并的时候只要保证根节点加上原来的就行了. 
            printf("%d\n", size[find(a)]);
        }
    }
}

#include <bits/stdc++.h>
using namespace std;

const int N = 100005;
int h[N], ne[N], e[N], idx;

void insert(int x){
  int k = (x%N + N) % N;
  e[idx] = x;
  ne[idx] = h[k];
  h[k]=idx++;
}

bool find(int x){
  int k = (x%N+N)%N;
  for(int i = h[k]; i!= -1; i = ne[i]){
    if(e[i] == x) return 1;
  }
  return 0;
}

int main(){
  int n; 
  cin>>n;
  memset(h, -1, sizeof(h));
  while(n--){
    char op[2];
    int x;
    scanf("%s%d", op, &x);
    if(*op == 'I')insert(x);
    else {if(find(x)) cout<<"Yes"; else cout<<"No";cout<<"\n";}
  }
}
