A. 排序

我们可以把最大的放在一个连续区间里面, 因此就枚举每个区间, 看最大的取和就行了.

int a[MAXN];
void solve(){
    int n, k;
    cin>>n>>k;
    int ans = 0;
    F(i, 1, n) cin>>a[i];
    for(int i=1; i<=k; i++){
        int cur = i; 
        int mx = 0;
        while(cur<=n){
            // printf("a[%d] = %d\n", cur,a[cur]);

            mx = max(mx, a[cur]);
            cur += k;
        }
        // printf("Ans += %d\n", mx);
        ans += mx;
    }
    cout<<ans<<endl;
}
