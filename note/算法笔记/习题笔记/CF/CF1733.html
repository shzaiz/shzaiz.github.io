A. 排序

我们可以把最大的放在一个连续区间里面, 因此就枚举每个区间, 看最大的取和就行了.

int a[MAXN];
void solve(){
    int n, k;
    cin>>n>>k;
    int ans = 0;
    F(i, 1, n) cin>>a[i];
    for(int i=1; i<=k; i++){
        int cur = i; 
        int mx = 0;
        while(cur<=n){
            // printf("a[%d] = %d\n", cur,a[cur]);

            mx = max(mx, a[cur]);
            cur += k;
        }
        // printf("Ans += %d\n", mx);
        ans += mx;
    }
    cout<<ans<<endl;
}

B. 构造

每次会出来一个胜者, 一个输家. 并且总有一个输家的赢得个数是0. 因此就可以着手构造0 m这种情况就行了.

void solve(){
    int n, a, b; cin>>n>>a>>b;
    int mx = max(a, b);
    if(mx == 0) {
        printf("-1\n");
        return;
    }
    if(min(a, b) != 0 || (n-1)% mx ){
        printf("-1\n");
        return;
    }
    for(int i=2; i<=n; i+=max(a,b)){
        F(j, 1, max(a, b)){
            printf("%d ", i);
        } 
    }
    printf("\n");
}

C. 构造

最多 \(n\) 次, 肯定用满就可以构造了. 因为 \(l\) 和 \(r\) 是可以交换的, 让他们都相等也不是难事.

int a[MAXN];
void solve(){
    int n; cin>>n;
    F(i, 1, n) cin>>a[i];
    
    printf("%d\n", n-1);
    if(n>1) printf("1 %d\n", n);

    int par = (a[1]+a[n])%2 ? a[1]:a[n];
    F(i, 2, n-1){
        if((par+a[i])%2) printf("1 %d\n", i);
        else printf("%d %d\n", i,n);
    }


}
