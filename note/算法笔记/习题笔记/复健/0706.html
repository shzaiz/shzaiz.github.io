bool solve(int np){
    int ans=0;
    F(i,1,n-1) {
        int diff = a[i+1]-a[i]; 
        if(diff >= np){
            ans += floor((double)diff/(double)np); 
            if(diff%np==0) ans-=1;
        }
    }
    if(ans>k) return 0; else return 1;
}
