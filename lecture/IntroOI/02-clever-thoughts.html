long long n, m, k;

double calc(double p){
    double ans = 0, res = 1.0+p;
    for(int i=1; i<=k; i++){
        ans += m/res;
        res *= (1.0+p);
    }
    return ans;
}

double query(double le, double ri){
    for(int i=0; i<10000; i++){
        double mid = (le+ri)/2

        if(hm < (double)n){
            ri = mid;
        }else if(hm > (double)n){
            le = mid;
        }else{
            return mid;
        }
    }
    return mid;
}

#include <cstdio>
#include <algorithm>

const int inf=1e9;
int n,m,a[100001];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    int l=0,r=inf,now,tot;
    while(l<r){
        int mid=(l+r+1)>>1;
        now=a[1];
        tot=1;
        for(int i=2;i<=n;++i){
            if(a[i]-now>=mid){now=a[i],++tot;}
            if(tot>=m)break;
        }
        if(tot<m)r=mid-1;
        else l=mid;
    }
    cout<<l<<endl;
}

	double ans=f(x,1);

#define fog(i,x) for(int i = head[x];i;i = e[i].nxt)

	e[++cnt].to = v;

int dep[N], fa[N][20]; 
void dfs1(int u,int f){
	fa[u][0] = f;
	dep[u] = dep[f ]+1;
	for(int i = 1;i<20;i++){
		fa[u][i] = fa[fa[u][i-1]][i-1];
	}
	fog(i,u){
		int v = e[i].to;

int lca(int u,int v){
	if(dep[u]>dep[v]) swap(u,v); 
	for(int i =19;i>=0;i--){
		if(dep[u]>dep[fa[v][i]] || fa[v][i]==0) continue;
		v = fa[v][i];

		if(fa[u][i] == fa[v][i] || fa[u][i]==0 || fa[v][i]==0) continue;

	}

}
